(** * Coproduct of natural transformations *)
Require Import Category.Core Functor.Core Category.Sum Functor.Sum NaturalTransformation.Core.

Set Universe Polymorphism.
Set Implicit Arguments.
Generalizable All Variables.
Set Asymmetric Patterns.

Section sum.
  
C, C', D:PreCategory
F, G:Functor C D
F', G':Functor C' D
T:NaturalTransformation F G
T':NaturalTransformation F' G'
NaturalTransformation (F + F') (G + G')

  
C, C', D:PreCategory
F, G:Functor C D
F', G':Functor C' D
T:NaturalTransformation F G
T':NaturalTransformation F' G'
NaturalTransformation (F + F') (G + G')

    
C, C', D:PreCategory
F, G:Functor C D
F', G':Functor C' D
T:NaturalTransformation F G
T':NaturalTransformation F' G'
forall (s d : (C + C')%category)
(m : morphism (C + C') s d),
(match
   d as x
   return
     (morphism D ((F + F') _0 x)%object
        ((G + G') _0 x)%object)
 with
 | Datatypes.inl c => T c
 | Datatypes.inr c' => T' c'
 end o (F + F') _1 m)%morphism =
((G + G') _1 m
 o match
     s as x
     return
       (morphism D ((F + F') _0 x)%object
          ((G + G') _0 x)%object)
   with
   | Datatypes.inl c => T c
   | Datatypes.inr c' => T' c'
   end)%morphism

    abstract (
        repeat (intros [] || intro); simpl;
        auto with natural_transformation
      ).
  Defined.
End sum.
