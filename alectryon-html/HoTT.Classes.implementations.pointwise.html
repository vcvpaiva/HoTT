  Instance top_fun : Top (A -> B)
    := fun _ => ‚ä§.

  Instance join_fun : Join (A -> B) :=
    fun (f g : A -> B) (a : A) => (f a) ‚äî (g a).

  Instance meet_fun : Meet (A -> B) :=
    fun (f g : A -> B) (a : A) => (f a) ‚äì (g a).

  (** Try to solve some of the lattice obligations automatically *)
  Create HintDb lattice_hints.
  #[local]
  Hint Resolve
       associativity
       absorption
       commutativity | 1 : lattice_hints.
  Ltac reduce_fun := compute; intros; apply path_forall; intro.
  Ltac try_solve_fun :=
    reduce_fun;
    eauto 10 with lattice_hints typeclass_instances.
