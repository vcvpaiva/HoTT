Definition ab_biprod_corec {A B X : AbGroup}

  : X $-> ab_biprod A B := grp_prod_corec f g.

Definition ab_corec_beta {X Y A B : AbGroup} (f : X $-> Y) (g0 : Y $-> A) (g1 : Y $-> B)
