(** ** Precomposition with an essentially surjective functor is faithful *)
Section faithfull_precomposition_essential_surjective.
  (** Quoting the HoTT Book:

  Lemma. If [A], [B], [C] are precategories and [H : A ‚Üí B] is an
  essentially surjective functor, then [(‚Äì ‚àò H) : (B ‚Üí C) ‚Üí (A ‚Üí C)]
  is faithful. *)

  Context `{fs : Funext}.
  Variables A B C : PreCategory.

  Variable H : Functor A B.

  Context `{H_is_essentially_surjective : IsEssentiallySurjective A B H}.

  
fs:Funext
A, B, C:PreCategory
H:Functor A B
H_is_essentially_surjective:IsEssentiallySurjective
  H
F, G:Functor B C
T, U:NaturalTransformation F G
a:A
b:B
f:H _0 a <~=~> b
H':T oR H = U oR H
T b = U b

H':forall x : A,
T (H _0 x)%object = U (H _0 x)%object
T b = U b

    
fs:Funext

G _1 f o U (H _0 a)%object o (F _1 f)^-1

G _1 f o U (H _0 a)%object o (F _1 f)^-1

    reflexivity.
  Qed.
