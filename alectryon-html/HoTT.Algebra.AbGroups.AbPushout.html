(** * Pushouts of abelian groups. *)

Definition ab_pushout {A B C : AbGroup}
           (f : A $-> B) (g : A $-> C) : AbGroup
  := QuotientAbGroup
       (ab_biprod B C)
       (grp_image (ab_biprod_corec f (g $o ab_homo_negation))).
